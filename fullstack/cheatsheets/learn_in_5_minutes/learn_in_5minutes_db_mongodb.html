▶

Installation

sudo apt install mongodb mongodb-clients mongodb-server mongo-tools

▶

Connect

mongo

▶

Show databases

show dbs

▶

Connect to db

use cars

▶

Insert in collection

db.cars.insert({ _id:1, name: "Audi", price: 52642})
db.cars.insert({ _id:2, name: "Mercedes", price: 57127})
db.cars.insert({ _id:3, name: "Skoda", price: 9000})
db.cars.insert({ _id:4, name: "Volvo", price: 29000})

▶

Show collections

show collections

▶

Select all rows

db.cars.find()
db.cars.find({})

▶

Count rows

db.cars.find().count()

▶

Search

db.cars.find({ "name" : "Audi" })
db.cars.find({ price : { $gt: 4888 } })
db.cars.find().limit(5)

▶

Update

db.cars.updateOne({"name":"Audi"},{$set: {"price":10000}})

▶

Delete

db.cars.deleteOne({"_id": 4})

▶

Export

mongoexport --db cars --collection cars --out cars.json
mongoexport --db cars --collection cars --type=csv --fields name,price --out cars.csv

▶

Import

mongoimport --db cars --collection cars --file cars.json
mongoimport -d cars -c cars --type csv --file cars.csv --headerline

▶

Exit

exit

▶

Remove collection

db.cars.remove({})

▶

Drop collection

db.cars.drop()

▶

Drop DB

use cars
db.dropDatabase()
