Here is an example of how to use Spleeter with the 4stems model in Python to separate audio into four components: vocals, drums, bass, and other.
Code Example:
Python# Import necessary modules
from spleeter.separator import Separator
import os

def separate_audio(input_audio_path, output_directory):
    """
    Separates an audio file into 4 stems (vocals, drums, bass, other) using Spleeter.

    Args:
        input_audio_path (str): Path to the input audio file.
        output_directory (str): Directory to save the separated stems.

    Returns:
        None
    """
    try:
        # Initialize Spleeter with the 4stems model
        separator = Separator('spleeter:4stems')

        # Perform source separation
        separator.separate_to_file(input_audio_path, output_directory)

        print(f"Separation completed! Stems saved in: {output_directory}")
    except Exception as e:
        print(f"An error occurred: {e}")

if __name__ == "__main__":
    # Input audio file path (replace with your file path)
    input_audio = "path/to/your/audio_file.mp3"

    # Output directory for separated stems
    output_dir = "path/to/output_directory"

    # Ensure output directory exists
    os.makedirs(output_dir, exist_ok=True)

    # Call the function to separate audio
    separate_audio(input_audio, output_dir)
