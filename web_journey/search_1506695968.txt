nmcli device show enp3s0 | grep IP4.DNS  # Ubuntu >= 15 OK
nmcli dev show | grep DNS # OK
nmcli dev show | grep DNS | sed 's/\s\s*/\t/g' | cut -f 2

nmcli dev list iface enp3s0 | grep IP4   # Ubuntu <= 14
