gridSize = 3
temperature = 4
superconditionFactor = 16
topK = 128

def generateImageByPrompt(promt: str, gridSize=gridSize):
    image = model.generate_image(
    text=promt,
    seed=-1,
    temperature=temperature,
    top_k=int(topK),
    supercondition_factor=superconditionFactor,
    grid_size=gridSize            
    )
    
    return image

def extract_with_limit(zip_path, extract_path, max_bytes):
    with zipfile.ZipFile(zip_path, 'r') as z:
        for file_info in z.infolist():
            # Check if file size is within limits
            if file_info.file_size <= max_bytes:
                z.extract(file_info, extract_path)
            else:
                print(f"Skipping {file_info.filename}: Too large ({file_info.file_size} bytes)")

# Example: Limit to 10MB
extract_with_limit('img_align_celeba.zip', 'input/celeba-dataset/img_align_celeba/img_align_celeba', 10 * 1024 * 1024)
